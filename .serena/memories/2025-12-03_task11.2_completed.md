const logger = createLogger({ name: 'MyService' });
logger.info('Operation completed', { userId: '123' });

// 監査ログ
logger.audit({
  action: 'user.login',
  userId: 'user-123',
  resource: 'auth',
  details: { ip: '127.0.0.1' }
});

// パフォーマンス計測
const result = await logger.time('db-query', async () => {
